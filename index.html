function updateGraph() { const selectedValue = document.getElementById('doprava-select').value; let traces = []; if (selectedValue === "ALL") { const druhyDopravy = [...new Set(globalData.map(item => item.vuz_txt))]; traces = druhyDopravy.map(druh => createTrace(druh, false)); } else { traces = [createTrace(selectedValue, true)]; } const layout = { title: { text: selectedValue === "ALL" ? 'Srovnání přepravních výkonů' : `Detail: ${selectedValue}`, font: { size: 18 } }, autosize: true, height: 550, hovermode: 'x unified', xaxis: { title: 'Období', gridcolor: '#eee', tickangle: -45 }, yaxis: { title: 'mil. osobokilometrů', gridcolor: '#eee' }, legend: { orientation: 'h', y: -0.3 }, margin: { t: 60, b: 120, l: 80, r: 40 } }; Plotly.newPlot('myGraph', traces, layout, { responsive: true, displayModeBar: false }); } function createTrace(druh, fillEnabled) { // Seřazení dat podle období, aby čára nebyla zmatená const filtered = globalData .filter(item => item.vuz_txt === druh) .sort((a, b) => String(a.casobd_txt).localeCompare(String(b.casobd_txt))); return { x: filtered.map(d => d.casobd_txt), y: filtered.map(d => d.hodnota), name: druh, type: 'scatter', mode: 'lines+markers', fill: fillEnabled ? 'tozeroy' : 'none', line: { width: 3 }, marker: { size: 6 }, hovertemplate: '%{y} mil. os.-km' }; }
